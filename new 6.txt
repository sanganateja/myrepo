pipeline {
    agent any 
    stages {
        stage('Compile stage') { 
            steps {
			  withMaven(maven : mvn_home) {
			     sh 'mvn clean compile'
				 }         
            }
        }
        stage('Test stage') { 
            steps {
			  withMaven(maven : mvn_home) {
			     sh 'mvn test'
                }
            }
        }
        stage('Deploy stage') { 
            steps {
                  withMaven(maven : mvn_home) {
			     sh 'mvn deploy' 
            }
        }
    }
}
